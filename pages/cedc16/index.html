# 216. 组合总和 III

  • 题目：216. 组合总和 III (opens new window)

  • 讲解：文章讲解 (opens new window)、视频讲解 (opens new window)

  • 提示：如果把组合问题理解了，本题就容易一些了

# 录前想法

接上上一题的思路，修改收集节点的方法，就 AC 啦

以下是代码实现，已 AC (✔️)

class Solution {
private:
    vector<int> path;
    vector<vector<int>> result;

    int getSum(vector<int> &path){
        
        int sum = 0;

        for(int num : path) sum += num;

        return sum;
    }

    void backtracking(int k, int n, int startIndex){

        // 当树枝元素和 == n时收集结果
        if(getSum(path) == n && path.size() == k){

            result.push_back(path);
            return;
        }

        // 递归回溯
        for( int i = startIndex; i <= 9; ++i ){

            path.push_back(i);

            backtracking(k,n,i+1);

            path.pop_back();
        }

    }

public:
    vector<vector<int>> combinationSum3(int k, int n) {

        path.clear();
        result.clear();

        backtracking(k,n,1);

        return result;

    }
};

# 录后想法

  1. 可以每次 减少n ，来判断 n == 0 来优化 getSum () 函数的调用
class Solution {
private:
    vector<int> path;
    vector<vector<int>> result;

    void backtracking(int k, int n, int startIndex){

        // 当收集的树枝 == k时再判断
        if(path.size() == k && n == 0){

            result.push_back(path);
            return;
        }

        // 递归回溯
        for( int i = startIndex; i <= 9; ++i ){

            path.push_back(i);
            // n - i 传入的时副本不是指针所以不会对原来的n有影响
            // 也就是不会回溯n
            backtracking(k,n - i,i + 1);
            path.pop_back();

        }

    }

  1. 剪枝优化

    1. 限定的元素个数 （上一题的结论）【 （ k - path.size () ）】

      1. 元素的个数：path.size ()
      2. 需要的个数：k - path.size ()
      3. 从哪里开始：n - (k - path.size () ) + 1

    2. 当 元素 > 和 时就不用再遍历了

优化后的代码实现如下：

void backtracking(int k, int n, int startIndex){

        // 当收集的树枝 == k时再判断
        if(path.size() == k && n == 0){

            result.push_back(path);
            return;
        }

        // 递归回溯
        for( int i = startIndex; i <= 9 - (k - path.size()) + 1; ++i ){

            path.push_back(i);
            
            // 和已经超出了n
            // 也就是减少减少过了头，所以变成了负数
            // 执行剪枝操作
            if(n < 0){
                path.pop_back();
                return;
            } 
            
            backtracking(k,n - i,i + 1);
            path.pop_back();

        }
}

class Solution {
private:
    string path;
    vector<string> result;

    const string letterMap[10] = {
        "", // 0
        "", // 1
        "abc", // 2
        "def", // 3
        "ghi", // 4
        "jkl", // 5
        "mno", // 6
        "pqrs", // 7
        "tuv", // 8
        "wxyz", // 9
    };

    void backtracking(string digits,int index){

        if(index == digits.size()){

            result.push_back(path);
            return;
        }

        // 循环遍历
        int digit = digits[index] - '0';
        string letters = letterMap[digit];
        for(int i = 0; i < letters.size(); ++i){

            path.push_back(letters[i]);
            backtracking(digits,index+1);
            path.pop_back();
        }

    }
public:
    vector<string> letterCombinations(string digits) {
        
        path.clear();
        result.clear();
        
        if(digits.size() > 0) backtracking(digits,0);

        return result;
    }
};
