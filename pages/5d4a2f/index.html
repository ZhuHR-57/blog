# 454. 四数相加 II

  • 题目：454. 四数相加 II (opens new window)

  • 讲解：文章讲解 (opens new window)、视频讲解 (opens new window)

  • 要求：map 巧解

# 录前想法

​ 做过，记得思路，两个为一组记录为 MAP 的 Key，再两个一组记录为 MAP 的 - Key

​ 下面是实现代码，已 AC (✔️)

class Solution {
public:
    int fourSumCount(vector<int>& nums1, vector<int>& nums2, vector<int>& nums3, vector<int>& nums4) {

        // 创建map
        unordered_map<int,int> map;
        
        int ans = 0;

        // 第一组循环
        for(int num1 : nums1){
            for(int num2 : nums2){
                map[num1+num2]++;
            }
        }

        // 第二组循环
        for(int num3 : nums3){
            for(int num4 : nums4){
                ans += map[-(num3+num4)];
            }
        }

        return ans;
    }
};

# 383. 赎金信

  • 题目：383. 赎金信 (opens new window)

  • 讲解：文章讲解 (opens new window)

  • 要求：和上一题同思路

# 录前想法

​ 与上一题思路一致，已 AC (✔️)

class Solution {
public:
    bool canConstruct(string ransomNote, string magazine) {

        // 创建map
        unordered_map<char,int> map;

        // 放入ransomNote
        for(int i = 0; i < ransomNote.size(); ++i){
            map[ransomNote[i]]++;
        }

        // 出magazine
        for(int i = 0; i < magazine.size(); ++i){
            map[magazine[i]]--;
        }

        for(int i = 0; i < ransomNote.size(); ++i){
            if(map[ransomNote[i]] > 0){
                return false;
            }
        }

        // 判断是否全部出完
        return true;
    }
};

# 录后想法

map 的空间消耗要比数组大一些的，因为 map 要维护红黑树或者哈希表，而且还要做哈希函数，是费时的

使用数组会节省空间

class Solution {
public:
    bool canConstruct(string ransomNote, string magazine) {

        // 创建map
        int arr[26] = {0};

        // 放入ransomNote
        for(int i = 0; i < magazine.size(); ++i){
            arr[magazine[i] - 'a']++;
        }

        // 出magazine
        for(int i = 0; i < ransomNote.size(); ++i){
            arr[ransomNote[i] - 'a']--;

            if(arr[ransomNote[i] - 'a'] < 0){
                return false;
            }
        }

        // 判断是否全部出完
        return true;
    }
};

以下是最终实现的代码，已 AC (✔️)：

class Solution {
public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        
        // 创建二位数组结果集合
        vector<vector<int>> result;
        int left,right;

        // 排序
        sort(nums.begin(), nums.end());

        // 遍历
        for(int i = 0; i < nums.size(); ++i){

            if(nums[i] > 0){return result;}

            // 对i去重
            if(i > 0 && nums[i] == nums[i-1]){continue;}

            // 创建双指针
            left = i + 1;
            right = nums.size() - 1;

            while(left < right){
                
                // 求和
                int tmp = nums[i] + nums[left] + nums[right];
                
                if(tmp > 0){// 三数之和 > 0
                    // right--
                    right--;
                }else if(tmp < 0){// 三数之和 < 0
                    // left++
                    left++;
                }else{// 三数之和 = 0
                    // 收集结果
                    result.push_back(vector<int>{nums[i],nums[left],nums[right]});

                    // 对left去重
                    while(left < right && nums[left] == nums[left + 1]){
                        left++;
                    }

                    // 对right去重
                    while(left < right && nums[right] == nums[right - 1]){
                        right--;
                    }

                    // 同时缩小
                    left++;
                    right--;
                } 
            }
        }
            
        // 返回结果集合
        return result;
    }
};

# 18. 四数之和

  • 题目：18. 四数之和 (opens new window)
  • 讲解：文章讲解 (opens new window)、视频讲解 (opens new window)
  • 要求：
      1. 为什么 454. 四数相加 II 会简单很多？
      2. 双指针

# 录前想法

和三数之和的思路相似，学习完三数之和有思路写出代码，已 AC (✔️)

class Solution {
public:
    vector<vector<int>> fourSum(vector<int>& nums, int target) {

        int size = nums.size();     // 获取数组长度
        vector<vector<int>> result; // 结果集合
        long tmpSum = 0;             // 是数之和
        int left,right;             // 双指针

        // 小于4判断
        if (size < 4) { return result; }

        // 排序
        sort(nums.begin(), nums.end());

        // 循环a
        for(int a = 0; a < size - 3; ++a){
            
            // 对a去重
            if(a > 0 && nums[a] == nums[a-1]){continue;}

            // 循环b
            for(int b = a + 1; b < size - 2; ++b){

                // 对b去重
                if(b > a + 1 && nums[b] == nums[b-1]){continue;}

                // 双指针
                left = b + 1;
                right = size - 1;

                // 双指针移动
                while(left < right){

                    // 求和
                    tmpSum = (long)nums[a]+nums[b]+nums[left]+nums[right];

                    // 对比target
                    if(tmpSum > target){
                        --right;
                    }else if(tmpSum < target){
                        ++left;
                    }else{
                        result.push_back(vector<int>{nums[a],nums[b],nums[left],nums[right]});

                        // left 去重
                        while(left < right && nums[left] == nums[left+1]){++left;}

                        // right去重
                        while(left < right && nums[right] == nums[right-1]){--right;}

                        ++left;
                        --right;
                    }
                }
            }
        }

        return result;
    }
};

# 录后想法

剪枝的细节操作，叹服

整理完后，写出以下代码，已 AC (✔️)

  • 求和时 long 类型，防止溢出
class Solution {
public:
    vector<vector<int>> fourSum(vector<int>& nums, int target) {

        int size = nums.size();     // 获取数组长度
        vector<vector<int>> result; // 结果集合
        long tmpSum = 0;             // 是数之和 long防止溢出
        int left,right;             // 双指针

        // 小于4判断
        if (size < 4) { return result; }

        // 排序
        sort(nums.begin(), nums.end());

        // 循环a
        for(int a = 0; a < size - 3; ++a){
            
            // 对a去重
            if(a > 0 && nums[a] == nums[a-1]){continue;}

            // 循环b
            for(int b = a + 1; b < size - 2; ++b){

                // 对b去重
                if(b > a + 1 && nums[b] == nums[b-1]){continue;}

                // 一层剪枝
                if(nums[a]+nums[b] > 0 && target > 0 && nums[a]+nums[b]>target){break;}
                // 双指针
                left = b + 1;
                right = size - 1;

                // 双指针移动
                while(left < right){

                    // 求和
                    tmpSum = (long)nums[a]+nums[b]+nums[left]+nums[right];

                    // 对比target
                    if(tmpSum > target){
                        --right;
                    }else if(tmpSum < target){
                        ++left;
                    }else{
                        result.push_back(vector<int>{nums[a],nums[b],nums[left],nums[right]});

                        // left 去重
                        while(left < right && nums[left] == nums[left+1]){++left;}

                        // right去重
                        while(left < right && nums[right] == nums[right-1]){--right;}

                        ++left;
                        --right;
                    }
                }
            }
        }

        return result;
    }
};
