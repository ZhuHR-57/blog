# 669. 修剪二叉搜索树

  • 题目：669. 修剪二叉搜索树 (opens new window)

  • 讲解：文章讲解 (opens new window)、视频讲解 (opens new window)

  • 提示：较难，比添加增加和删除节点难的多，建议先看视频理解

# 录前想法

和删除节点类似，从删除单个值变成了删除多个值

代码实现如下，未 AC (❌)

class Solution {
public:
    TreeNode* trimBST(TreeNode* root, int low, int high) {

        // 1. root为空
        if(root == nullptr) return root;

        // 2. 当在范围内
        if(root->val < low || root->val > high){

            // 1. 叶子节点
            if(root->left == nullptr && root->right == nullptr){
                return nullptr;
            }

            // 2. 左有右无
            if(root->left != nullptr && root->right == nullptr){
                return root->left;
            }

            // 3. 左无右有
            if(root->left == nullptr && root->right != nullptr){
                return root->right;
            }

            // 4. 左右都用
            if(root->left != nullptr && root->right != nullptr){
                
                // 临时节点保存右子树
                TreeNode *tmpNode = root->right;

                // 遍历找到右子树的最左节点
                while(tmpNode->left){
                    tmpNode = tmpNode->left;
                }

                // 右子树最左节点的左孩子，接上左子树
                tmpNode->left = root->left;

                // 删除节点
                return root->right;
            }
        }

        // 左递归
        root->left = trimBST(root->left,low,high);

        // 右递归
        root->right = trimBST(root->right,low,high);

        return root;
    }
};

# 录后想法

看了卡哥的思路后，发现和我的思路不一样

卡哥更加充分的利用递归

  1. 小于左区间

    1. 节点的左孩子 必然也小于左区间
    2. 节点的右孩子 可能不小于左区间 【难点就再删除小于的部分，答案是 递归 】

  2. 大于右区间

    1. 节点的右孩子 必然也大于右区间
    2. 节点的左孩子 可能不大于右区间 【难点就再删除大于的部分，答案是 递归 】

根据以上分析，代码实现如下，已 AC (✔️)

class Solution {
public:
    TreeNode* trimBST(TreeNode* root, int low, int high) {

        // 1. root为空
        if(root == nullptr) return root;
        // 2. 当节点小于左区间
        if(root->val < low){
            // 递归处理节点的右孩子,直到满足为止
            return trimBST(root->right,low,high);
        }
        // 3. 当节点大于又区间
        if(root->val > high){
            // 递归处理节点的左孩子,直到满足为止
            return trimBST(root->left,low,high);
        }
        // 左递归
        root->left = trimBST(root->left,low,high);
        // 右递归
        root->right = trimBST(root->right,low,high);

        return root;
    }
};

# 108. 将有序数组转换为二叉搜索树

  • 讲解：文章讲解 (opens new window)、视频讲解 (opens new window)

# 录前想法

有序刚好对应着二叉搜索树的中序遍历，可以对应着这顺序去构造

以下是代码实现，未 AC (❌)

class Solution {
private:
    int index = 0;
public:
    TreeNode* sortedArrayToBST(vector<int>& nums) {

        if(nums.size() == 0) return nullptr;

        TreeNode *root = new TreeNode(nums[(nums.size()/2)]);
        return tarversal(root,nums);

    }

    TreeNode* tarversal(TreeNode *root,vector<int>& nums){

        // 中序遍历
        if(root == nullptr) return new TreeNode(nums[index]);

        // 左
        root->left = tarversal(root->left,nums);

        // 中
        cout << root->val << endl;
        if(index < nums.size()){
            if(index != (nums.size()/2)){
                index++;
            }else{
                index += 2;
            }
        }

        // 右
        root->right = tarversal(root->left,nums);

        return root;
    }
};

# 录后想法

突然想到之前中序 + 前 / 后序构造二叉树的过程， 分割左右区间再构造

  1. 二叉树搜索树根节点 是 数组中间的元素
  2. 确定分割方式，[] or [)
  3. 递归分割

代码实现如下，已 AC (✔️)

class Solution {
private:
        TreeNode* tarversal(vector<int>& nums,int left,int right){

        // 左开右闭等于是不可能的
        // 左闭右闭等于表示最有一个元素，需要处理
        if(left >= right) return nullptr;
            
        // 防越界
        int mid = left + ((right - left) / 2);
        TreeNode *root = new TreeNode(nums[mid]);
        /*
            划分左右
        */
        // 左 [left,mid)
        root->left = tarversal(nums,left,mid);
        // 右 [mid + 1 , right)
        root->right = tarversal(nums,mid + 1,right);

        return root;
    }
public:
    TreeNode* sortedArrayToBST(vector<int>& nums) {
        
        return tarversal(nums,0,nums.size());
    }
};

# 538. 把二叉搜索树转换为累加树

  • 题目：538. 把二叉搜索树转换为累加树 (opens new window)

  • 讲解：文章讲解 (opens new window)、视频讲解 (opens new window)

  • 提示：双指针法

# 录前想法

  1. 先中序遍历一遍树，讲元素放入数组中
  2. 数组求和
  3. 再中序遍历一遍树，处理节点与和

实现代码如下，已 AC (✔️)

class Solution {

    vector<int> value;
    long long sum;

    // 中序遍历
    void traversal(TreeNode *root){

        if(root == nullptr) return;

        traversal(root->left);
        value.push_back(root->val);
        traversal(root->right);

    }
    // 求和
    void getSum(){
        for(int num : value){
            sum += num;
        }
    }
    // 中序遍历
    void traversalVlaue(TreeNode *root){


        if(root == nullptr) return;

        traversalVlaue(root->left);
        
        int tmp = root->val;
        root->val = sum;
        sum = sum - tmp;

        traversalVlaue(root->right);
        
    }

public:
    TreeNode* convertBST(TreeNode* root) {
        
        value.clear();
        sum = 0;
        TreeNode *node = root;

        traversal(root);
        getSum();
        traversalVlaue(node);

        return node;
    }
};

# 录后想法

oops! 反中序遍历，震惊‼️

核心 ： 前指针，保存上一个处理的节点

以下是代码实现，已 AC (✔️)

class Solution {

    TreeNode *pre;

    // 中序遍历
    void traversal(TreeNode *root){

        if(root == nullptr) return;
        // 右
        traversal(root->right);
        // 中
        if(pre != nullptr) root->val += pre->val;
        pre = root;
        // 左
        traversal(root->left);

    }
public:
    TreeNode* convertBST(TreeNode* root) {

        pre = nullptr;
        traversal(root);

        return root;
    }
};

# 总结

  • 讲解：文章讲解 (opens new window)

终于过完一遍二叉树的题目了😭，虽然前面写后面忘记，磕磕巴巴的，但总算都了解了有点🎇

  1. 二叉树的构造

无论普通二叉树还是二叉搜索树一定前序，都是先构造中节点

然后左右分割，构造左右子树

  1. 普通二叉树的属性

一般都是后序遍历

求深度和高度都可以用后序遍历

  1. 二叉搜索树的属性

一般都是中序遍历
