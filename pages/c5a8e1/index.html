# 1005.K 次取反后最大化的数组和

  • 题目：1005.K 次取反后最大化的数组和 (opens new window)

  • 讲解：文章讲解 (opens new window)、视频讲解 (opens new window)

  • 提示：本题简单一些，估计大家不用想着贪心 ，用自己直觉也会有思路

# 录前想法

  1. 先排序
  2. 从小到大反转 k-- index++
  3. 若 k 还有剩余
    1. 再排序
      1. 如果 k 是偶数，则反转最小值
      2. 是奇数则不变

以下是代码实现，未 AC (❌)

class Solution {
public:
    int getSum(vector<int>& nums){

        int sum = 0;
        for(int num : nums) sum += num;
        return sum;
    }

    int largestSumAfterKNegations(vector<int>& nums, int k) {

        // 排序
        sort(nums.begin(),nums.end());
        int index = 0;
        int numsLen = nums.size();

        // 就一位数
        if(nums.size() == 1){ return getSum(nums);}

        // 都是正数
        if(nums[0] >= 0){

            if(k%2 != 0) nums[0] = -nums[0];
            return getSum(nums);
        }

        // 有正有负
        while(k-- && index < numsLen){

            if(nums[index] <= 0){
                nums[index] = -nums[index];
                index++;
            }else{
                sort(nums.begin(),nums.end());
                
                if(nums[0] >= 0){
                    if(k%2 == 0) nums[0] = -nums[0];
                    return getSum(nums);
                }
            }
        }

        return getSum(nums);
    }
};

实现代码如下，已 AC (✔️)

class Solution {
static bool cmp(int a, int b) {
    return abs(a) > abs(b);
}
public:
    int largestSumAfterKNegations(vector<int>& nums, int k) {

        int numsLen = nums.size();
        int sum = 0;

        // 按照绝对值的大小排序
        sort(nums.begin(),nums.end(),cmp);

        // 非0数反转
        for(int i = 0; i < numsLen; ++i){
            if(nums[i] < 0 && k > 0){
                nums[i] *= -1;
                k--;
            }
        }

        // 奇数就再反转最小数
        if(k%2 != 0) nums[numsLen -1] *= -1; 

        // 求和
        for(int num : nums) sum += num;

        return sum;
    }
};

# 134. 加油站

  • 题目：134. 加油站 (opens new window)

  • 讲解：文章讲解 (opens new window)、视频讲解 (opens new window)

  • 提示：本题有点难度，不太好想，推荐大家熟悉一下方法二

# 录前想法

一个一个加油站判断

若从此点能出发能跑完一周，就返回此点位置

若不能返回 - 1

实现代码如下，未 AC (❌)

class Solution {

public:
    int canCompleteCircuit(vector<int>& gas, vector<int>& cost) {

        int index;

        for(index = 0; index < gas.size(); ++index){
            
            if(gas[index] > cost[index]){
                if(check(gas,cost,index)){
                    return index;
                }
            }
        }

        return -1;

    }

    bool check(vector<int>& gas, vector<int>& cost,int index){

        // 出发点
        int oil = gas[index];
        int gasLen = gas.size() - 1;
        int gIndex,cIndex;
        
        // 循环
        while(gasLen--){

            // 更新位置
            gIndex = index % gas.size();
            cIndex = (index + 1) % gas.size();
            // 扣油和加油
            oil = oil - cost[gIndex] + gas[cIndex];
            // 判断油量
            if(oil < 0) return false;
            // 更新index
            index++;
        } 

        // 最后一趟
        if(oil - cost[index % gas.size()] < 0) return false;

        return true;
    }
};

// 到1   7 - 3 + 2 = 6
// 到2   6 - 4 + 3 = 5
// 到3   5 - 5 = 0 刚好到

# 录后想法

只考虑油的 差值

  • 只要油（差值和）够往前跑就行
  • 如果此点不够，则表明前面的都不够，就从下一个点开始且重置差值和

思路如下图所示：

带和实现入下，已 AC (✔️)

class Solution {
public:
    int canCompleteCircuit(vector<int>& gas, vector<int>& cost) {
        int totalGas = 0; // 总油量
        int currentGas = 0; // 当前油量
        int startStation = 0; // 起始加油站索引

        for (int i = 0; i < gas.size(); ++i) {
            int gasDiff = gas[i] - cost[i]; // 当前加油站的油量差

            totalGas += gasDiff; // 累加总油量
            currentGas += gasDiff; // 更新当前油量

            if (currentGas < 0) {
                // 说明从前面的点出发都不够油，直接试试从后面出发
                startStation = i + 1;
                // 新的记录
                currentGas = 0;
            }
        }

        // 如果总油量小于零，说明无法绕环路行驶一周
        // 否则，起始加油站就是解
        return (totalGas < 0) ? -1 : startStation;
    }
};

# 135. 分发糖果

  • 讲解：文章讲解 (opens new window)、视频讲解 (opens new window)

  • 提示：本题涉及到一个思想，就是想处理好一边再处理另一边，不要两边想着一起兼顾，后面还会有题目用到这个思路

# 录前想法

如果不相等，那肯定要差值，直接 candy++

class Solution {
public:
    int candy(vector<int>& ratings) {

        int childsNum = ratings.size();
        if(childsNum == 1) return 1;

        int candySum = childsNum;
        for(int i = 1; i < childsNum; ++i){
            if(ratings[i] != ratings[i-1]) candySum++;
        }

        return candySum; 
    }
};

整体代码如下，已 AC (✔️)

class Solution {
public:
    int candy(vector<int>& ratings) {

        // 保存第一次比较的糖果数
        vector<int> candys(ratings.size(),1); // 初始化为1

        // 右 > 左
        for(int i = 1; i < ratings.size(); ++i){
            if(ratings[i] > ratings[i-1]){
                candys[i] = candys[i-1] + 1;
            }
        }

        // 右 < 左
        // 从倒数第二个元素开始比较
        for(int i = ratings.size() - 2; i >= 0; --i){
            if(ratings[i] > ratings[i+1]){
                candys[i] = max(candys[i+1]+1,candys[i]);
            }
        }

        // 求和
        int sum = 0;
        for(int num : candys) sum += num;

        return sum;
    }
};
