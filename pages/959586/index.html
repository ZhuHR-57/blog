bool isValid(vector<string> &chessboard,int row,int col,int n){

    // 检查同列
    for(int i = 0; i < row; ++i){
        if(chessboard[i][col] == 'Q'){
            return false;
        }
    }    

    // 检查同斜线 45度
    // 后退一层，往回看
    for(int i = row - 1, j = col - 1; i >=0 && j >= 0; i--, j--){
        if(chessboard[i][j] == 'Q'){
            return false;

    // 检查同斜线 135度

    for(int i = row - 1,j = col + 1; i >= 0 && j < n;i--,j++){

        if(chessboard[i][j] == 'Q') return false;
    }              

    return true;
}

整体代码实现如下：

class Solution {
    vector<vector<string>> result;

    // chess 二维棋盘数组
    // n 边界
    // row 遍历层数
    void backTracking(vector<string> &chessboard,int n,int row){

        // 当遍历到第n行，就收集结果
        if(row == n){
            result.push_back(chessboard);
            return;
        }

        // 单层遍历
        // col = 0 尝试暴力放置到每一个位置
        for(int col = 0; col < n; ++col){

            // 如果位置合法
            if(isValid(chessboard,row,col,n)){
                // 放置棋子
                chessboard[row][col] = 'Q';
                // 递归
                backTracking(chessboard,n,row+1);
                // 回溯
                chessboard[row][col] = '.';
            }
        }
    }

    // 验证棋子是否合法
    // row col 行 列 row col 都是会大一级
    // 不能同行
    // 不能同列
    // 不能同斜线 （45度和135度角）
    bool isValid(vector<string> &chessboard,int row,int col,int n){

        // 检查同列
        for(int i = 0; i < row; ++i){
            if(chessboard[i][col] == 'Q'){
                return false;
            }
        }    

        // 检查同斜线 45度
        // 后退一层，往回看
        for(int i = row - 1, j = col - 1; i >=0 && j >= 0; i--, j--){
            if(chessboard[i][j] == 'Q'){
                return false;
            }
        }                    

        // 检查同斜线 135度
        // 后退一层，往回看
        for(int i = row - 1,j = col + 1; i >= 0 && j < n;i--,j++){
            
            if(chessboard[i][j] == 'Q') return false;
        }              

        return true;
    }

public:
    vector<vector<string>> solveNQueens(int n) {

        vector<string> chessboard(n, string(n, '.'));
        result.clear();

        backTracking(chessboard,n,0);

        return result;
    }
};

# 分割

难点在于 分割线 ，分割线就是在需要分割的元素后面一位，如： a| "|" 就是 a 的分割线

所以，分割时的终止条件是 原数组下标+1

核心三要素：

  1. 分割的范围： [startIndex , i] ，startIndex 是前一个元素的分割线

  2. 结束条件： startIndex == s.size() ，当指向最后最后一个元素是， startIndex = i + 1 ，所以是等于 s.size()

  3. 如何截取字串

    // [startIndex,i]在s中的子串
string str = s.substr(startIndex, i - startIndex + 1);
    1
    2

# startIndex 的用法总结

  1. 递归时 i + 1 ，层树遍历中 不使用 上一个已使用的数
backtracking(n,k,i + 1);
1

  1. 递归时就是 i ，层数遍历中 使用 上一个已使用的数
backtracking(n,k,i);
1

  1. 层树遍历时，允许有重复元素，但不能用重复的组合
if(i > startIndex && candidates[i] == candidates[i-1]) continue;
1

去除重复元素， 排序后 ，遍历时 跳过 前一个相同元素

  • 删除数组中的重复元素： i > 0 && arr[i] == arr[i-1]
  • 删除一层中的重复元素： i > startIndex && arr[i] == arr[i-1]
