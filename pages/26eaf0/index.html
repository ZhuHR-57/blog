  • 讲解：文章讲解 (opens new window)、视频讲解 (opens new window)

  • 要求：理解什么时候用库函数，什么时候不用库函数

# 录前想法

交换首末元素，注意数组越界，已 AC (✔️)

class Solution {
public:
    void reverseString(vector<char>& s) {

        int tmp;
        int len = s.size() - 1;

        // 交换
        for(int i = 0; i <= len/2; ++i){
            
            tmp = s[i];
            s[i] = s[len-i];
            s[len-i] = tmp;

        }
    }
};

# 录前想法

有思路，但就是写不对，提交错了三次😓

以下是我的代码，未 AC (❌)

class Solution {
public:
    string reverseStr(string s, int k) {
        
        int len = s.size();
        int pos = 0;

        if(len <= k){
            reverse(s,0 ,len - 1);
            return s;
        }

        for(pos = k - 1; pos < len; pos += 2 * k){

            // 反转前k个字符
            // abcdefghe 8
            reverse(s,pos - k + 1 ,pos);
            
        }

        // 如果剩余字符少于 k 个，则将剩余字符全部反转。
        // abcdefg 8
        if(len - pos < k){
            reverse(s,pos,len - pos - 1);
        }

        // 小于 2k 但大于或等于 k 个，则反转前 k 个字符
        if(len - pos < 2 * k && len - pos >= k){
            reverse(s,pos,pos + k - 1);
        }

        return s;
    }

    // 反转
    void reverse(string &s,int l,int r){
        for(;l < r;l++,r--){
            swap(s[l],s[r]);
        }
    }
};

# 录后想法

哇，感觉自己写的好复杂😂

有考虑到间断跳跃

以下是整理后已 AC 的代码（✔️）

class Solution {
public:
    string reverseStr(string s, int k) {

        int len = s.size();

        for (int i = 0; i < len; i += (2 * k)) {
            
            if(i+k<=len){
                reverse(s,i,i+k-1);
                continue;
            }

            reverse(s,i,len-1);
        }

        return s;
    }

    // 反转
    void reverse(string &s,int l,int r){
        for(;l < r;l++,r--){
            swap(s[l],s[r]);
        }
    }
};

# 剑指 Offer 05. 替换空格

  • 题目：剑指 Offer 05. 替换空格 (opens new window)

  • 讲解：文章讲解 (opens new window)

  • 建议：对于线性数据结构，填充或者删除，后序处理会高效的多。好好体会一下。

# 录前想法

创建新的 char 数组，数组容量为 原数组+Space * 2 来保存结果集

class Solution {
public:
    string replaceSpace(string s) {

        // 统计空格数量
        int len = s.size();
        int spaceCount = 0;
        int pos = 0;
        
        for(int i = 0; i < len; ++i){
            if(s[i] == ' '){
                ++spaceCount;
            }
        }


        // 拓容
        char ans[len + 2 * spaceCount];

        // 循环填充
        for(int i = 0; i < len; ++i){
            
            if(s[i] == ' '){
                ans[pos++] = '%';
                ans[pos++] = '2';
                ans[pos++] = '0';
            }else{
                ans[pos++] = s[i];
            }
            
        }

        return string(ans);
    }
};
