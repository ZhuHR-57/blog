  • 讲解：文章讲解 (opens new window)、视频讲解 (opens new window)

  • 要求：理解什么时候用库函数，什么时候不用库函数

# 录前想法

交换首末元素，注意数组越界，已 AC (✔️)

class Solution {
public:
    void reverseString(vector<char>& s) {

        int tmp;
        int len = s.size() - 1;

        // 交换
        for(int i = 0; i <= len/2; ++i){
            
            tmp = s[i];
            s[i] = s[len-i];
            s[len-i] = tmp;

        }
    }
};

# 541. 反转字符串 II

  • 题目：541. 反转字符串 II (opens new window)

  • 讲解：文章讲解 (opens new window)、视频讲解 (opens new window)

  • 要求：先独立做再看解析

# 录前想法

有思路，但就是写不对，提交错了三次😓

以下是我的代码，未 AC (❌)

class Solution {
public:
    string reverseStr(string s, int k) {
        
        int len = s.size();
        int pos = 0;

        if(len <= k){
            reverse(s,0 ,len - 1);
            return s;
        }

        for(pos = k - 1; pos < len; pos += 2 * k){

            // 反转前k个字符
            // abcdefghe 8
            reverse(s,pos - k + 1 ,pos);
            
        }

        // 如果剩余字符少于 k 个，则将剩余字符全部反转。
        // abcdefg 8
        if(len - pos < k){
            reverse(s,pos,len - pos - 1);
        }

        // 小于 2k 但大于或等于 k 个，则反转前 k 个字符
        if(len - pos < 2 * k && len - pos >= k){
            reverse(s,pos,pos + k - 1);
        }

        return s;
    }

    // 反转
    void reverse(string &s,int l,int r){
        for(;l < r;l++,r--){
            swap(s[l],s[r]);
        }
    }
};

# 剑指 Offer 05. 替换空格

  • 题目：剑指 Offer 05. 替换空格 (opens new window)

  • 讲解：文章讲解 (opens new window)

  • 建议：对于线性数据结构，填充或者删除，后序处理会高效的多。好好体会一下。

# 录前想法

创建新的 char 数组，数组容量为 原数组+Space * 2 来保存结果集

class Solution {
public:
    string replaceSpace(string s) {

        // 统计空格数量
        int len = s.size();
        int spaceCount = 0;
        int pos = 0;
        
        for(int i = 0; i < len; ++i){
            if(s[i] == ' '){
                ++spaceCount;
            }
        }


        // 拓容
        char ans[len + 2 * spaceCount];

        // 循环填充
        for(int i = 0; i < len; ++i){
            
            if(s[i] == ' '){
                ans[pos++] = '%';
                ans[pos++] = '2';
                ans[pos++] = '0';
            }else{
                ans[pos++] = s[i];
            }
            
        }

        return string(ans);
    }
};

# 录后想法

  • 双指针
  • resize()

class Solution {
public:
    string replaceSpace(string s) {

        // 统计空格数量
        int oldLen = s.size();
        int newLen;
        int spaceCount = 0;
        
        for(int i = 0; i < oldLen; ++i){
            if(s[i] == ' '){
                ++spaceCount;
            }
        }

        newLen = oldLen + 2 * spaceCount;

        // 拓容
        s.resize(newLen);

        // 循环填充
        // fast 指向原数组的最后一个位置
        // slow 指向新数组的最后一个位置
        for(int fast = oldLen - 1,slow = newLen - 1; fast >= 0;){
            
            if(s[fast] == ' '){
                s[slow--] = '0';
                s[slow--] = '2';
                s[slow--] = '%';
                fast--;
            }else{
                s[slow--] = s[fast--];
            }
        }

        return s;
    }
};

# 151. 翻转字符串里的单词

  • 题目：151. 翻转字符串里的单词 (opens new window)

  • 讲解：文章讲解 (opens new window)、视频讲解 (opens new window)

  • 建议：这道题目基本把 刚刚做过的字符串操作 都覆盖了，不过就算知道解题思路，本题代码并不容易写，要多练一练。

# 录前想法

  1. 去除多余的空格
  2. 整体反转
  3. 单词反转

已经 AC (✔️)

class Solution {
public:
    string reverseWords(string s) {
        
        int len = s.size();

        // 1.双指针移动删除空格
        // 对空格去重
        int fast = 0,slow = 0;
        // 1.1 删除
        // 找到第一个不为空格的元素
        while(s[fast] == ' ' && fast < len){fast++;}
        // 1.2 中间遇到空格
        while(fast < len){
            
            if(s[fast] != ' ' || (s[fast] == ' ' && s[fast-1] != ' ' && fast != 0)){
                s[slow++] = s[fast++];
            }else{
                while(s[fast] == ' ' && fast < len){fast++;}
            }

        }
        // 1.3 结尾遇到空格
        if(s[slow - 1] == ' '){
            s.resize(slow - 1);
            len = slow - 1;
        }else{
            s.resize(slow);
            len = slow;
        }

        // 2. 整体反转
        reverse(s,0,len-1);

        // 3. 再对每个单词反转
        int l,r;
        for(l = 0,r = 0; r < len; ++r){

            if(s[r] == ' '){
                reverse(s,l,r - 1);
                l = r + 1;
            }
        }

        // 反转最后一个单词
        reverse(s,l,r - 1);

        return s;
    }

    // 反转
    void reverse(string &str,int l,int r){
        for(;l<r; ++l,--r){
            swap(str[l],str[r]);
        }
    }
};

# 剑指 Offer58-II. 左旋转字符串

  • 题目：剑指 Offer58-II. 左旋转字符串 (opens new window)

  • 讲解：文章讲解 (opens new window)

  • 建议：题解中的解法如果没接触过的话，应该会想不到

# 录前想法

  1. 整体反转
  2. 倒数 k 个字符反转
  3. 正数 len - k 个字符反转

以下代码已 AC (✔️)

class Solution {
public:
    string reverseLeftWords(string s, int n) {

        // abcdefg
        // gfedc ba
        // cdefg ab

        // lrloseumgh
        // hgmu esolrl
        // umgh lrlose

        int len = s.size();
        // 整体反转
        reverse(s,0,len-1);

        // 倒数n个反转
        reverse(s,len-n,len-1);

        // 0 ~ len - n 反转
        reverse(s,0,len-n-1);

        return s;
    }

    void reverse(string &str,int l,int r){
        for(;l<r;++l,--r){
            swap(str[l],str[r]);
        }
    }
};
